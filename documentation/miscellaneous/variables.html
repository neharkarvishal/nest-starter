Default value : (user: IUser) => { const firstNameLetter = user.firstName ? user.firstName.charAt(0).toUpperCase() : '' if (firstNameLetter) { return getDummyImage(330, 300, firstNameLetter) } const firstCityLetter = user.email.charAt(0).toUpperCase() return getDummyImage(330, 300, firstCityLetter) }
